import matplotlib.pyplot as plt import pandas as pd df = pd.read_excel("vendas.xlsx") # Criar figura com 2x2 subplots fig, axes = plt.subplots(2, 2, figsize=(15, 10)) fig.suptitle('Dashboard de Vendas', fontsize=18, fontweight='bold') # Gr√°fico 1: Barras axes[0, 0].bar(df['Produto'], df['Quantidade'], color='steelblue') axes[0, 0].set_title('Quantidade Vendida') axes[0, 0].set_xlabel('Produto') axes[0, 0].tick_params(axis='x', rotation=45) axes[0, 0].grid(axis='y', alpha=0.3) # Gr√°fico 2: Barras horizontais axes[0, 1].barh(df['Produto'], df['Pre√ßo'], color='coral') axes[0, 1].set_title('Pre√ßo por Produto') axes[0, 1].set_xlabel('Pre√ßo (R$)') axes[0, 1].grid(axis='x', alpha=0.3) # Gr√°fico 3: Pizza df_total = df['Quantidade'] * df['Pre√ßo'] axes[1, 0].pie(df_total, labels=df['Produto'], autopct='%1.1f%%') axes[1, 0].set_title('Distribui√ß√£o do Faturamento') # Gr√°fico 4: Linha axes[1, 1].plot(df['Produto'], df['Quantidade'], marker='o', linewidth=2, color='green') axes[1, 1].set_title('Tend√™ncia de Vendas') axes[1, 1].tick_params(axis='x', rotation=45) axes[1, 1].grid(True, alpha=0.3) plt.tight_layout() plt.savefig('dashboard.png', dpi=300) plt.show()
